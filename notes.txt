@Composable
fun TabMediaPlayer(
    state: EngineState,
    mediaPlayer: MediaPlayer
) {
    val context = LocalContext.current
    var rawAudioEnergy by remember { mutableFloatStateOf(0f) }
    
    var hasAudioPermission by remember { 
        mutableStateOf(ContextCompat.checkSelfPermission(context, Manifest.permission.RECORD_AUDIO) == PackageManager.PERMISSION_GRANTED) 
    }
    
    val permissionLauncher = rememberLauncherForActivityResult(
        contract = ActivityResultContracts.RequestPermission()
    ) { isGranted ->
        hasAudioPermission = isGranted
    }
    
    val smoothedEnergy by animateFloatAsState(
        targetValue = rawAudioEnergy,
        animationSpec = tween(durationMillis = 120), 
        label = "audio_smoothing"
    )

    LaunchedEffect(smoothedEnergy, state.isPlaying) {
        if (state.isPlaying && hasAudioPermission) {
            state.amplitude = 20f + (smoothedEnergy * 280f) 
            state.flowSpeed = 1.0f + (smoothedEnergy * 1.5f)
        } else {
            state.amplitude = 8f
            state.flowSpeed = 0.15f 
        }
    }

    val filePickerLauncher = rememberLauncherForActivityResult(
        contract = ActivityResultContracts.GetContent()
    ) { uri: Uri? ->
        uri?.let { selectedUri ->
            try {
                if (mediaPlayer.isPlaying) mediaPlayer.stop()
                mediaPlayer.reset()
                
                mediaPlayer.setDataSource(context, selectedUri)
                mediaPlayer.prepare()
                
                state.mediaDuration = mediaPlayer.duration
                state.currentPosition = 0
                state.isPlaying = false
                rawAudioEnergy = 0f
                
                var trackName = "Unknown Track"
                context.contentResolver.query(selectedUri, null, null, null, null)?.use { cursor ->
                    val nameIndex = cursor.getColumnIndex(OpenableColumns.DISPLAY_NAME)
                    if (cursor.moveToFirst() && nameIndex != -1) trackName = cursor.getString(nameIndex)
                }
                
                val mmr = MediaMetadataRetriever()
                mmr.setDataSource(context, selectedUri)
                
                val title = mmr.extractMetadata(MediaMetadataRetriever.METADATA_KEY_TITLE)
                val artist = mmr.extractMetadata(MediaMetadataRetriever.METADATA_KEY_ARTIST)
                if (!title.isNullOrEmpty()) {
                    trackName = if (!artist.isNullOrEmpty()) "$artist - $title" else title
                }
                state.currentTrackName = trackName

                val artBytes = mmr.embeddedPicture
                if (artBytes != null) {
                    val bitmap = BitmapFactory.decodeByteArray(artBytes, 0, artBytes.size)
                    state.mediaAlbumArt = bitmap.asImageBitmap()
                    
                    Palette.from(bitmap).maximumColorCount(128).clearFilters().generate { palette ->
                        val swatches = palette?.swatches ?: emptyList()
                        val bestSwatch = swatches.maxByOrNull { swatch ->
                            val hsl = swatch.hsl
                            if (hsl[1] < 0.15f || hsl[2] < 0.1f || hsl[2] > 0.9f) -1f else (hsl[1] * 1000f) + (hsl[2] * 500f)
                        }
                        val targetColorInt = bestSwatch?.rgb ?: palette?.vibrantSwatch?.rgb ?: android.graphics.Color.WHITE
                        state.extractedMediaColor = Color(targetColorInt)
                    }
                } else {
                    state.mediaAlbumArt = null
                    state.extractedMediaColor = Color(0xFFFFFFFF)
                }
                mmr.release()
            } catch (e: Exception) { e.printStackTrace() }
        }
    }

    DisposableEffect(state.isPlaying, hasAudioPermission) {
        var visualizer: Visualizer? = null
        if (state.isPlaying && hasAudioPermission) {
            try {
                visualizer = Visualizer(mediaPlayer.audioSessionId)
                visualizer.captureSize = Visualizer.getCaptureSizeRange()[1] 
                visualizer.setDataCaptureListener(object : Visualizer.OnDataCaptureListener {
                    override fun onWaveFormDataCapture(visualizer: Visualizer?, waveform: ByteArray?, samplingRate: Int) {
                        waveform?.let {
                            var sum = 0.0
                            for (byte in it) {
                                val amplitude = (byte.toInt() and 0xFF) - 128
                                sum += (amplitude * amplitude).toDouble()
                            }
                            val rms = sqrt(sum / it.size)
                            val normalizedEnergy = (rms / 128.0).coerceIn(0.0, 1.0).toFloat()
                            val finalEnergy = (normalizedEnergy * 1.5f).coerceAtMost(1.0f)
                            
                            // Safely route the background audio data to the UI thread
                            Handler(Looper.getMainLooper()).post {
                                rawAudioEnergy = finalEnergy
                            }
                        }
                    }
                    override fun onFftDataCapture(visualizer: Visualizer?, fft: ByteArray?, samplingRate: Int) {}
                }, Visualizer.getMaxCaptureRate() / 2, true, false)
                visualizer.enabled = true
            } catch (e: Exception) { e.printStackTrace() }
        } else {
            rawAudioEnergy = 0f
        }
        onDispose { visualizer?.enabled = false; visualizer?.release() }
    }

    val uiAlpha by animateFloatAsState(targetValue = if (state.isMediaUiVisible) 1f else 0f, animationSpec = tween(300), label = "uiAlpha")
    val uiOffsetY by animateDpAsState(targetValue = if (state.isMediaUiVisible) 0.dp else 50.dp, animationSpec = tween(300), label = "uiOffset")

    Column(
        modifier = Modifier
            .fillMaxWidth()
            .offset(y = uiOffsetY)
            .graphicsLayer { alpha = uiAlpha },
        verticalArrangement = Arrangement.spacedBy(8.dp)
    ) {
        // FIX: Pure if statement removes the node cleanly without exit animations
        if (!hasAudioPermission) {
            Card(colors = CardDefaults.cardColors(containerColor = MaterialTheme.colorScheme.errorContainer)) {
                Column(modifier = Modifier.fillMaxWidth().padding(16.dp), horizontalAlignment = Alignment.CenterHorizontally) {
                    Text("Visualizer Disabled", fontWeight = FontWeight.Bold, color = MaterialTheme.colorScheme.onErrorContainer)
                    Spacer(modifier = Modifier.height(4.dp))
                    Text("Android requires microphone access to read the internal audio track.", style = MaterialTheme.typography.bodySmall, color = MaterialTheme.colorScheme.onErrorContainer, textAlign = androidx.compose.ui.text.style.TextAlign.Center)
                    Spacer(modifier = Modifier.height(12.dp))
                    Button(
                        onClick = { permissionLauncher.launch(Manifest.permission.RECORD_AUDIO) },
                        colors = ButtonDefaults.buttonColors(containerColor = MaterialTheme.colorScheme.error)
                    ) {
                        Text("Grant Permission")
                    }
                }
            }
        }

        Card(colors = CardDefaults.cardColors(containerColor = MaterialTheme.colorScheme.surface.copy(alpha = 0.9f))) {
            Column(modifier = Modifier.fillMaxWidth().padding(12.dp), horizontalAlignment = Alignment.CenterHorizontally) {
                Row(verticalAlignment = Alignment.CenterVertically) {
                    Text("Extracted: ", style = MaterialTheme.typography.labelMedium)
                    Spacer(modifier = Modifier.width(8.dp))
                    Box(modifier = Modifier.size(24.dp).background(state.extractedMediaColor, CircleShape).border(1.dp, Color.White, CircleShape))
                }
                Spacer(modifier = Modifier.height(12.dp))
                Text("Test Palettes:", style = MaterialTheme.typography.labelSmall, color = Color.Gray)
                Spacer(modifier = Modifier.height(8.dp))
                Row(modifier = Modifier.fillMaxWidth(), horizontalArrangement = Arrangement.SpaceEvenly) {
                    val testColors = listOf(Color(0xFF00FF00), Color(0xFFFF0055), Color(0xFF00DDFF), Color(0xFFFF5500), Color(0xFFAA00FF))
                    testColors.forEach { color ->
                        Box(modifier = Modifier.size(32.dp).background(color, CircleShape).border(if (state.extractedMediaColor == color) 2.dp else 0.dp, Color.White, CircleShape).clickable { state.extractedMediaColor = color })
                    }
                }
            }
        }

        Card(colors = CardDefaults.cardColors(containerColor = MaterialTheme.colorScheme.surface.copy(alpha = 0.9f))) {

                    value = if (state.mediaDuration > 0) state.currentPosition.toFloat() / state.mediaDuration.toFloat() else 0f,

                        state.currentPosition = newPos
                        try { mediaPlayer.seekTo(newPos) } catch(e: Exception) {}
                    },
                    modifier = Modifier.fillMaxWidth()

                Row(modifier = Modifier.fillMaxWidth(), horizontalArrangement = Arrangement.SpaceEvenly) {
                    Button(
                        onClick = { try { mediaPlayer.start(); state.isPlaying = true; state.fluidVolumeTarget = 1f } catch(e: Exception) {} },

        targetValue = activeColorTarget,
        animationSpec = tween(durationMillis = 1500, easing = LinearOutSlowInEasing),
        label = "color_crossfade"
    )

    Box(
        modifier = Modifier
            .fillMaxSize()
            .pointerInput(Unit) {
                awaitPointerEventScope {
                    while (true) {
                        val event = awaitPointerEvent(PointerEventPass.Initial)
                        val change = event.changes.firstOrNull()
                        if (change != null) {
                            if (state.fluidState == FluidState.IDLE) {
                                state.touchX = change.position.x
                                state.touchY = change.position.y
                            }
                            state.isScreenTouched = change.pressed
                        }
                    }
                }
            }
            .pointerInput(Unit) {
                detectTapGestures(
                    onTap = { 
                        if (state.selectedCategory == 0) {
                            state.isNotificationActive = true 
                        } else if (state.selectedCategory == 2) {
                            state.isMediaUiVisible = !state.isMediaUiVisible
                        }
                    }, 
                    onDoubleTap = { state.isNotificationActive = false }
                )
            }
    ) {
        AmbientEdgeLighting(
            modifier = Modifier.fillMaxSize(),
            shaderString = activeShaderString,
            cornerRadius = 120f, appColor = animatedAppColor,
            speed = state.flowSpeed, thickness = state.thickness, amplitude = state.amplitude, noiseScale = state.noiseScale,
            masterAlpha = masterAlpha,
            pillCenterY = pillCenterYPx, pillWidth = pillWidthPx, pillHeight = pillHeightPx,
            gravityX = fluidMomentumX, gravityY = fluidMomentumY,
            touchX = activeTouchX, touchY = activeTouchY, touchActive = magneticTouchStrength,
            fluidVolume = fluidVolume,
            transferDirection = state.transferDirection,
            effectSize = state.effectSize
        ) {
            // FIX 1: Safe Background Crossfade
            Crossfade(
                targetState = state.selectedCategory == 2 && state.mediaAlbumArt != null,
                animationSpec = tween(800),
                label = "bg_crossfade"
            ) { isMediaBg ->
                if (isMediaBg) {
                    Box(
                        modifier = Modifier.fillMaxSize().background(Color.Black),
                        contentAlignment = Alignment.Center
                    ) {
                        Image(
                            bitmap = state.mediaAlbumArt!!,
                            contentDescription = "Album Art",
                            contentScale = ContentScale.Fit,
                            modifier = Modifier.fillMaxWidth(0.85f).aspectRatio(1f).clip(RoundedCornerShape(16.dp))
                        )
                    }
                } else {
                    Image(
                        painter = painterResource(id = android.R.drawable.sym_def_app_icon), 
                        contentDescription = "Background", 
                        contentScale = ContentScale.Crop, 
                        modifier = Modifier.fillMaxSize()
                    )
                }
            }
        }

        if (state.selectedCategory == 0) {
            Card(
                modifier = Modifier.align(Alignment.TopCenter).offset(y = pillOffsetY).alpha(masterAlpha).fillMaxWidth(0.9f).height(64.dp),
                shape = RoundedCornerShape(32.dp), colors = CardDefaults.cardColors(containerColor = MaterialTheme.colorScheme.surface.copy(alpha = 0.95f)),
                elevation = CardDefaults.cardElevation(defaultElevation = 0.dp) 
            ) {}
        }

        Column(
            modifier = Modifier
                .align(Alignment.BottomCenter)
                .fillMaxWidth()
                .padding(bottom = 24.dp, start = 16.dp, end = 16.dp),
            verticalArrangement = Arrangement.spacedBy(16.dp)
        ) {
            
            Box(modifier = Modifier.fillMaxHeight(0.55f)) {
                // FIX 2: LayoutNode Tab Clamping fix
                // Each tab now isolates its own layout tree during the fade transition
                Crossfade(
                    targetState = state.selectedCategory, 
                    animationSpec = tween(400),
                    label = "tab_crossfade"
                ) { category ->
                    Column(
                        modifier = Modifier.fillMaxSize().verticalScroll(rememberScrollState()),
                        verticalArrangement = Arrangement.spacedBy(8.dp)
                    ) {
                        when (category) {
                            0 -> TabNotifications(state)
                            1 -> TabSendReceive(state, screenWidthPx)
                            2 -> TabMediaPlayer(state, mediaPlayer)
                        }
                    }
                }
            }
            
            val isTabBarVisible = state.selectedCategory != 2 || state.isMediaUiVisible
            val tabBarAlpha by animateFloatAsState(targetValue = if (isTabBarVisible) 1f else 0f, animationSpec = tween(300), label = "tabAlpha")
            val tabBarOffsetY by animateDpAsState(targetValue = if (isTabBarVisible) 0.dp else 100.dp, animationSpec = tween(300), label = "tabOffset")

            Card(
                modifier = Modifier.offset(y = tabBarOffsetY).graphicsLayer { alpha = tabBarAlpha },
                colors = CardDefaults.cardColors(containerColor = MaterialTheme.colorScheme.surface.copy(alpha = 0.9f)),
                elevation = CardDefaults.cardElevation(defaultElevation = 8.dp)
            ) {
                TabRow(selectedTabIndex = state.selectedCategory, containerColor = Color.Transparent, indicator = { tabPositions -> TabRowDefaults.SecondaryIndicator(Modifier.tabIndicatorOffset(tabPositions[state.selectedCategory]), color = MaterialTheme.colorScheme.primary) }) {
                    Tab(selected = state.selectedCategory == 0, onClick = { 
                        state.selectedCategory = 0; state.isNotificationActive = true; state.fluidVolumeTarget = 1f; state.touchX = screenWidthPx / 2f; state.touchY = 0f 
                    }, text = { Text("Notifications", fontWeight = FontWeight.Bold, style = MaterialTheme.typography.labelMedium) })
                    Tab(selected = state.selectedCategory == 1, onClick = { 
                        state.selectedCategory = 1; state.isNotificationActive = false; state.fluidVolumeTarget = 0f 
                    }, text = { Text("Send/Receive", fontWeight = FontWeight.Bold, style = MaterialTheme.typography.labelMedium) })
                    Tab(selected = state.selectedCategory == 2, onClick = { 
                        state.selectedCategory = 2; state.isNotificationActive = false; state.fluidVolumeTarget = 1f 
                    }, text = { Text("Media", fontWeight = FontWeight.Bold, style = MaterialTheme.typography.labelMedium) })
                }
            }
        }
    }
}

@RequiresApi(Build.VERSION_CODES.TIRAMISU)
@Composable
fun AmbientEdgeLighting(
    modifier: Modifier = Modifier, shaderString: String, cornerRadius: Float = 120f,
    appColor: Color, speed: Float, thickness: Float, amplitude: Float, noiseScale: Float, masterAlpha: Float,
    pillCenterY: Float, pillWidth: Float, pillHeight: Float, gravityX: Float, gravityY: Float,
    touchX: Float, touchY: Float, touchActive: Float, fluidVolume: Float, transferDirection: Float,
    effectSize: Float,
    content: @Composable () -> Unit
) {
    var time by remember { mutableFloatStateOf(0f) }
    var wavePhase by remember { mutableFloatStateOf(0f) }
    
    LaunchedEffect(Unit) {
        val startTime = System.nanoTime()
        var lastTime = startTime
        while (true) {
            awaitFrame()
            val currentTime = System.nanoTime()
            val dt = (currentTime - lastTime) / 1_000_000_000f
            lastTime = currentTime
            
            time = (currentTime - startTime) / 1_000_000_000f
            wavePhase += (dt * speed * 15f) * fluidVolume
        }
    }

    val shader = remember(shaderString) { 
        RuntimeShader(shaderString.filter { it.code in 0..127 }) 
    }

    Box(
        modifier = modifier.graphicsLayer {
            shader.setFloatUniform("in_Resolution", size.width, size.height)
            shader.setFloatUniform("in_Time", time)
            shader.setFloatUniform("in_WavePhase", wavePhase) 
            shader.setFloatUniform("in_CornerRadius", cornerRadius)
            shader.setFloatUniform("in_AppColor", appColor.red, appColor.green, appColor.blue)
            shader.setFloatUniform("in_Speed", speed)
            shader.setFloatUniform("in_Spread", thickness)
            shader.setFloatUniform("in_Amplitude", amplitude)
            shader.setFloatUniform("in_NoiseScale", noiseScale)
            shader.setFloatUniform("in_MasterAlpha", masterAlpha) 
            shader.setFloatUniform("in_PillCenterY", pillCenterY)
            shader.setFloatUniform("in_PillWidth", pillWidth)
            shader.setFloatUniform("in_PillHeight", pillHeight)
            shader.setFloatUniform("in_Gravity", gravityX, gravityY)
            shader.setFloatUniform("in_TouchPos", touchX, touchY)
            shader.setFloatUniform("in_TouchActive", touchActive)
            shader.setFloatUniform("in_FluidVolume", fluidVolume)
            shader.setFloatUniform("in_TransferDirection", transferDirection)
            shader.setFloatUniform("in_EffectSize", effectSize)
            
            renderEffect = RenderEffect.createRuntimeShaderEffect(shader, "in_Background").asComposeRenderEffect()
            clip = true
        }
    ) {
        content()
    }
}
