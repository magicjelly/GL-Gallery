// ==========================================
// 1. AUDIO DSP ENGINE
// ==========================================
@UnstableApi
class AudioTrendEngine : TeeAudioProcessor.AudioBufferSink {
    var audioGain: Float = 25.0f 
    private val _macroBreath = MutableStateFlow(0f)
    val macroBreath: StateFlow<Float> = _macroBreath 
    private val _microPulse = MutableStateFlow(0f)
    val microPulse: StateFlow<Float> = _microPulse
    private val _spectralBalance = MutableStateFlow(0f)
    val spectralBalance: StateFlow<Float> = _spectralBalance
    private var prevSample = 0f
    private var currentSpectralRatio = 0.5f

    override fun flush(sampleRateHz: Int, channelCount: Int, encoding: Int) {
        _macroBreath.value = 0f; _microPulse.value = 0f; _spectralBalance.value = 0.5f
    }

    override fun handleBuffer(buffer: ByteBuffer) {
        val limit = buffer.limit()
        if (limit == 0) return
        var sumSq = 0.0
        var highFreqSumSq = 0.0
        buffer.position(0)
        while (buffer.position() < limit - 1) {
            val raw = buffer.short.toFloat() / Short.MAX_VALUE
            val highFreq = raw - prevSample
            prevSample = raw
            sumSq += (raw * raw)
            highFreqSumSq += (highFreq * highFreq)
        }
        val totalRms = sqrt(sumSq / (limit / 2)).toFloat()
        val highFreqRms = sqrt(highFreqSumSq / (limit / 2)).toFloat()
        val energy = (1.0f - exp(-totalRms * audioGain)).coerceIn(0f, 1f)
        val balance = if (totalRms > 0.01f) (highFreqRms / totalRms).coerceIn(0f, 1f) else 0.5f
        
        currentSpectralRatio = (currentSpectralRatio * 0.95f) + (balance * 0.05f)
        _spectralBalance.value = currentSpectralRatio
        _macroBreath.value = (_macroBreath.value * 0.9f) + (energy * 0.1f)
        _microPulse.value = energy
    }
}

// ==========================================
// 2. AGSL SHADER: ROTATIONAL ADVECTION ACRYLIC
// ==========================================
const val EDGE_LIT_ACRYLIC_SRC = """
    uniform float2 u_resolution;
    uniform float u_macroBreath; 
    uniform float u_microPulse;  
    uniform float u_kineticTime;
    uniform float u_spectralBalance;
    uniform float2 u_boxSizePx;
    uniform float u_cornerRadiusPx;
    
    uniform float3 u_mutedColor;
    uniform float3 u_dominantColor;
    uniform float3 u_vibrantColor;

    float2 hash22(float2 p) {
        float3 p3 = fract(float3(p.xyx) * float3(.1031, .1030, .0973));
        p3 += dot(p3, p3.yzx + 33.33);
        return fract((p3.xx+p3.yz)*p3.zy);
    }

    float noise(float2 x) {
        float2 i = floor(x);
        float2 f = fract(x);
        f = f * f * (3.0 - 2.0 * f);
        return mix(
            mix(dot(hash22(i + float2(0.0, 0.0)), f - float2(0.0, 0.0)),
                dot(hash22(i + float2(1.0, 0.0)), f - float2(1.0, 0.0)), f.x),
            mix(dot(hash22(i + float2(0.0, 1.0)), f - float2(0.0, 1.0)),
                dot(hash22(i + float2(1.0, 1.0)), f - float2(1.0, 1.0)), f.x), f.y);
    }

    float fbm(float2 p) {
        float value = 0.0; float amplitude = 0.5;
        for (int i = 0; i < 3; i++) {
            value += amplitude * noise(p);
            p *= 2.0; amplitude *= 0.5;
        }
        return value;
    }

    float sdRoundRect(float2 p, float2 b, float r) {
        float2 d = abs(p) - b + float2(r);
        return min(max(d.x, d.y), 0.0) + length(max(d, 0.0)) - r;
    }

    float3 mixLinear(float3 colA, float3 colB, float t) {
        return sqrt(mix(colA * colA, colB * colB, clamp(t, 0.0, 1.0)));
    }

    half4 main(float2 fragCoord) {
        float2 uv = fragCoord / u_resolution;
        float aspect = u_resolution.x / u_resolution.y;
        float2 aspectUv = float2(uv.x * aspect, uv.y);
        float2 center = float2(0.5 * aspect, 0.5);
        
        // Anchored Origin
        float2 toPixel = aspectUv - center;
        
        // Hardware Binding
        float2 boxHalfSize = max((u_boxSizePx / u_resolution.y) * 0.5, float2(0.001));
        float cornerRadius = max(u_cornerRadiusPx / u_resolution.y, 0.001);
        
        float d = sdRoundRect(toPixel, boxHalfSize, cornerRadius);
        
        // --- 1. ROTATIONAL CURL DYNAMICS (THE PIVOT) ---
        float complexity = 1.0 + (u_spectralBalance * 2.0);
        float2 noiseUv = toPixel * complexity;
        float flow = u_kineticTime * 0.8;
        
        // Swirl Vector: Traverses the noise domain in a continuous circle around the UI.
        // This restores the continuous rolling whites of the previous version without diagonal drift.
        float2 swirlVector = normalize(float2(-toPixel.y, toPixel.x) + float2(0.001));
        float2 advect = swirlVector * flow;
        
        float2 e = float2(0.03, 0.0);
        float nx = fbm(noiseUv + e.xy + advect) - fbm(noiseUv - e.xy + advect);
        float ny = fbm(noiseUv + e.yx + advect) - fbm(noiseUv - e.yx + advect);
        float2 curl = float2(ny, -nx);
        
        // --- 2. ACRYLIC FRACTURE ---
        float warpStrength = 0.1 + (u_microPulse * 0.2);
        float2 fracturedPixel = toPixel + (curl * warpStrength);
        
        float shatteredSDF = abs(sdRoundRect(fracturedPixel, boxHalfSize, cornerRadius));
        float edgeGlow = 0.003 / (shatteredSDF + 0.001);
        
        float organicBreakup = fbm(toPixel * 5.0 + u_kineticTime);
        float causticTension = edgeGlow * organicBreakup * (1.0 + u_microPulse * 4.0);
        
        // --- 3. PURE ADDITIVE LIGHTING ---
        float opticalFade = exp(-max(d, 0.0) * (3.5 - u_macroBreath * 1.5));
        float3 deepAcrylic = mixLinear(u_mutedColor, u_dominantColor, opticalFade * 0.9);
        float3 ignitedEdges = u_vibrantColor * causticTension * opticalFade;
        
        float3 finalColor = deepAcrylic + ignitedEdges;
        
        // Hardware Mask & Priming
        finalColor *= smoothstep(-0.005, 0.0, d);
        finalColor = max(finalColor, float3(0.004)); 
        finalColor += float3((hash22(fragCoord).x - 0.5) * 0.006);
        finalColor = finalColor / (1.0 + finalColor * 0.2);
        
        return half4(clamp(finalColor, 0.0, 1.0), 1.0);
    }
"""

// ==========================================
// 3. UI ARCHITECTURE (STRICT 320dp LOCK)
// ==========================================
@OptIn(UnstableApi::class)
@Composable
fun ChromodynamicAcrylicEngine() {
    val context = LocalContext.current
    val audioEngine = remember { AudioTrendEngine() }
    val exoPlayer = remember { setupExoPlayer(context, audioEngine) }
    
    var currentArtBitmap by remember { mutableStateOf<android.graphics.Bitmap?>(null) }
    var mutedColor by remember { mutableStateOf(Color(0xFF0A0A0A)) }
    var dominantColor by remember { mutableStateOf(Color(0xFF1E1E1E)) }
    var vibrantColor by remember { mutableStateOf(Color(0xFF888888)) }
    
    var kineticTime by remember { mutableFloatStateOf(0f) }
    var smoothedPulse by remember { mutableFloatStateOf(0f) } 
    var pulseVelocity by remember { mutableFloatStateOf(0f) }
    var lastFrameNanos by remember { mutableLongStateOf(0L) }
    
    var isPlaying by remember { mutableStateOf(false) }
    var artSizePx by remember { mutableStateOf(IntSize.Zero) }
    
    val targetSizeDp = 320.dp
    val targetCornerRadiusDp = 36.dp
    val cornerRadiusPx = with(LocalDensity.current) { targetCornerRadiusDp.toPx() }
    val opticalShader = remember { RuntimeShader(EDGE_LIT_ACRYLIC_SRC) }

    DisposableEffect(exoPlayer) {
        val listener = object : Player.Listener {
            override fun onIsPlayingChanged(p: Boolean) { isPlaying = p }
            override fun onMediaMetadataChanged(m: MediaMetadata) {
                m.artworkData?.let { data ->
                    val bitmap = BitmapFactory.decodeByteArray(data, 0, data.size)
                    currentArtBitmap = bitmap
                    Palette.from(bitmap).generate { p ->
                        val fallback = 0xFF444444.toInt()
                        val dom = p?.getDominantColor(fallback) ?: fallback
                        val mut = p?.getDarkMutedColor(dom) ?: dom
                        val vib = p?.getLightVibrantColor(p.getVibrantColor(dom)) ?: dom
                        
                        mutedColor = Color(mut)
                        dominantColor = Color(dom)
                        vibrantColor = Color(vib)
                    }
                }
            }
        }
        exoPlayer.addListener(listener)
        onDispose { exoPlayer.release() }
    }

    val filePicker = rememberLauncherForActivityResult(ActivityResultContracts.GetContent()) { uri ->
        uri?.let { exoPlayer.setMediaItem(MediaItem.fromUri(it)); exoPlayer.prepare(); exoPlayer.play() }
    }

    val macroBreath by audioEngine.macroBreath.collectAsState()
    val spectralBalance by audioEngine.spectralBalance.collectAsState()

    LaunchedEffect(Unit) {
        while (true) {
            withFrameNanos { frameNanos ->
                if (lastFrameNanos == 0L) lastFrameNanos = frameNanos
                val dt = ((frameNanos - lastFrameNanos) / 1_000_000_000f).coerceIn(0f, 0.033f)
                lastFrameNanos = frameNanos
                
                val activeAlpha = if (isPlaying) 1f else 0f
                val targetPulse = audioEngine.microPulse.value
                val stiffness = 60f
                val damping = 0.9f * 2f * sqrt(stiffness) 
                
                val force = stiffness * (targetPulse - smoothedPulse) - damping * pulseVelocity
                pulseVelocity += force * dt
                smoothedPulse += pulseVelocity * dt
                
                kineticTime += dt * (0.2f + (smoothedPulse.coerceAtLeast(0f) * 1.5f)) * activeAlpha
            }
        }
    }

    Box(modifier = Modifier.fillMaxSize().background(Color.Black)) {
        
        // LAYER 1: AGSL Background
        Canvas(modifier = Modifier.fillMaxSize()) {
            opticalShader.setFloatUniform("u_resolution", size.width, size.height)
            opticalShader.setFloatUniform("u_kineticTime", kineticTime)
            opticalShader.setFloatUniform("u_macroBreath", macroBreath)
            opticalShader.setFloatUniform("u_microPulse", smoothedPulse.coerceAtLeast(0f))
            opticalShader.setFloatUniform("u_spectralBalance", spectralBalance) 
            opticalShader.setFloatUniform("u_boxSizePx", artSizePx.width.toFloat(), artSizePx.height.toFloat())
            opticalShader.setFloatUniform("u_cornerRadiusPx", cornerRadiusPx)
            opticalShader.setFloatUniform("u_mutedColor", mutedColor.red, mutedColor.green, mutedColor.blue)
            opticalShader.setFloatUniform("u_dominantColor", dominantColor.red, dominantColor.green, dominantColor.blue)
            opticalShader.setFloatUniform("u_vibrantColor", vibrantColor.red, vibrantColor.green, vibrantColor.blue)
            
            drawRect(brush = ShaderBrush(opticalShader))
        }

        // LAYER 2: Hardware-Aligned Album Art
        if (currentArtBitmap != null) {
            Box(
                modifier = Modifier
                    .align(Alignment.Center)
                    .size(targetSizeDp), // STRICT 320dp LOCK
                contentAlignment = Alignment.Center
            ) {
                Image(
                    bitmap = currentArtBitmap!!.asImageBitmap(), 
                    contentDescription = null, 
                    modifier = Modifier
                        .fillMaxSize() 
                        .clip(RoundedCornerShape(targetCornerRadiusDp))
                        .onSizeChanged { artSizePx = it },
                    contentScale = ContentScale.Crop
                )
            }
        }

        // LAYER 3: Persistent Controls & Telemetry
        Column(
            modifier = Modifier
                .align(Alignment.BottomCenter)
                .fillMaxWidth()
                .background(Color(0xD9080808))
                .padding(24.dp)
        ) {
            Row(
                modifier = Modifier.fillMaxWidth(), 
                horizontalArrangement = Arrangement.SpaceBetween,
                verticalAlignment = Alignment.CenterVertically
            ) {
                Text("EDGE-LIT ACRYLIC", color = Color.White, style = MaterialTheme.typography.titleSmall)
                Row(horizontalArrangement = Arrangement.spacedBy(8.dp)) {
                    Button(
                        onClick = { if (isPlaying) exoPlayer.pause() else exoPlayer.play() },
                        colors = ButtonDefaults.buttonColors(containerColor = Color.DarkGray)
                    ) {
                        Text(if (isPlaying) "PAUSE" else "PLAY")
                    }
                    Button(
                        onClick = { filePicker.launch("audio/*") }, 
                        colors = ButtonDefaults.buttonColors(containerColor = Color(0xFF2E7D32))
                    ) {
                        Text("OPEN")
                    }
                }
            }
            Spacer(modifier = Modifier.height(16.dp))
            TelemetryBar("FRACTURE IGNITION", smoothedPulse.coerceAtLeast(0f), Color(0xFFFF5722))
            TelemetryBar("OPTICAL DENSITY", macroBreath, Color(0xFF00E5FF))
            TelemetryBar("SHEAR STRESS", spectralBalance, Color(0xFFAEEA00))
        }
    }
}

@Composable
fun TelemetryBar(label: String, value: Float, color: Color) {
    Row(verticalAlignment = Alignment.CenterVertically, modifier = Modifier.padding(vertical = 4.dp)) {
        Text(label, color = Color.Gray, style = MaterialTheme.typography.labelSmall, modifier = Modifier.width(160.dp))
        Canvas(modifier = Modifier.height(8.dp).weight(1f).clip(RoundedCornerShape(4.dp))) {
            drawRect(color = Color.DarkGray)
            drawRect(color = color, size = size.copy(width = size.width * value))
        }
    }
}

@UnstableApi
fun setupExoPlayer(context: Context, audioEngine: AudioTrendEngine): ExoPlayer {
    val teeProcessor = TeeAudioProcessor(audioEngine)
    val audioSink = DefaultAudioSink.Builder().setAudioProcessors(arrayOf(teeProcessor)).build()
    val renderersFactory = object : DefaultRenderersFactory(context) {
        override fun buildAudioSink(c: Context, f: Boolean, a: Boolean, e: Boolean) = audioSink
    }
    return ExoPlayer.Builder(context, renderersFactory).build().apply { repeatMode = Player.REPEAT_MODE_ONE }
}
