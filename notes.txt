// 1. AFFECTIVE DSP ENGINE (WITH AGC)
// ==========================================
@UnstableApi
class AudioTrendEngine : TeeAudioProcessor.AudioBufferSink {
    private var rollingRms: Float = 0.05f 
    
    private val _macroBreath = MutableStateFlow(0f)
    val macroBreath: StateFlow<Float> = _macroBreath 
    private val _microPulse = MutableStateFlow(0f)
    val microPulse: StateFlow<Float> = _microPulse
    private val _spectralBalance = MutableStateFlow(0f)
    val spectralBalance: StateFlow<Float> = _spectralBalance
    private val _volatility = MutableStateFlow(0f) 
    val volatility: StateFlow<Float> = _volatility
    private val _roughness = MutableStateFlow(0f)  
    val roughness: StateFlow<Float> = _roughness

    private var prevSample = 0f
    private var prevHighFreqRms = 0f
    private var currentSpectralRatio = 0.5f

    override fun flush(sampleRateHz: Int, channelCount: Int, encoding: Int) {
        _macroBreath.value = 0f; _microPulse.value = 0f; _spectralBalance.value = 0.5f
        _volatility.value = 0f; _roughness.value = 0f
    }

    override fun handleBuffer(buffer: ByteBuffer) {
        val limit = buffer.limit()
        if (limit == 0) return
        var sumSq = 0.0
        var highFreqSumSq = 0.0
        var zeroCrossings = 0
        var prevSign = 0f
        
        buffer.position(0)
        while (buffer.position() < limit - 1) {
            val raw = buffer.short.toFloat() / Short.MAX_VALUE
            val highFreq = raw - prevSample
            if (raw * prevSign < 0) zeroCrossings++
            prevSample = raw
            prevSign = raw
            sumSq += (raw * raw)
            highFreqSumSq += (highFreq * highFreq)
        }
        
        val totalRms = sqrt(sumSq / (limit / 2)).toFloat()
        val highFreqRms = sqrt(highFreqSumSq / (limit / 2)).toFloat()
        
        rollingRms = (rollingRms * 0.995f) + (totalRms * 0.005f)
        val dynamicGain = (1.0f / (rollingRms.coerceAtLeast(0.001f))) * 0.6f
        
        val energy = (1.0f - exp(-totalRms * dynamicGain)).coerceIn(0f, 1f)
        val balance = if (totalRms > 0.01f) (highFreqRms / totalRms).coerceIn(0f, 1f) else 0.5f
        
        val rawFlux = abs(highFreqRms - prevHighFreqRms)
        prevHighFreqRms = highFreqRms
        val rawZcr = zeroCrossings.toFloat() / (limit / 2).toFloat()
        
        currentSpectralRatio = (currentSpectralRatio * 0.99f) + (balance * 0.01f)
        _spectralBalance.value = currentSpectralRatio
        _macroBreath.value = (_macroBreath.value * 0.92f) + (energy * 0.08f) 
        _microPulse.value = energy 
        _volatility.value = (_volatility.value * 0.90f) + ((rawFlux * dynamicGain * 5f).coerceIn(0f, 1f) * 0.10f)
        _roughness.value = (_roughness.value * 0.95f) + (rawZcr * 0.05f)
    }
}

// 2. AGSL SHADER: THE AETHERIUM HALO
// ==========================================
const val AETHERIUM_HALO_SRC = """
    uniform float2 u_resolution;
    uniform float u_macroBreath; 
    uniform float u_microPulse;  
    uniform float u_kineticTime; 
    uniform float u_spectralBalance;
    uniform float u_volatility; 
    uniform float u_roughness;  
    uniform float2 u_boxSizePx;
    uniform float u_cornerRadiusPx;
    
    uniform float3 u_mutedColor;
    uniform float3 u_dominantColor;
    uniform float3 u_vibrantColor;

    // Strict SDF Anchoring
    float sdRoundRect(float2 p, float2 b, float r) {
        float2 d = abs(p) - b + float2(r);
        return min(max(d.x, d.y), 0.0) + length(max(d, 0.0)) - r;
    }

    half4 main(float2 fragCoord) {
        float2 center = u_resolution * 0.5;
        float2 p = fragCoord - center;
        
        float2 boxHalfSize = max(u_boxSizePx * 0.5, float2(0.001));
        float trueD = sdRoundRect(p, boxHalfSize, u_cornerRadiusPx);
        
        // Exclude internal album art perfectly.
        if (trueD < 0.0) return half4(0.0);
        
        float theta = atan(p.y, p.x);
        float3 finalColor = float3(0.01); // OLED Black floor
        
        // Continuous Kinetic Light Sweep.
        float lightSweepBase = u_kineticTime * 0.8;
        
        const float RINGS = 6.0;
        
        for(float i = RINGS; i > 0.0; i -= 1.0) {
            // Bass energy pushes the rings outward
            float spacing = 28.0 + (u_macroBreath * 12.0);
            
            // MicroPulse causes spatial shear (bounded displacement)
            float shear = u_microPulse * 8.0 * sin(theta * 4.0 + (i * 2.0));
            float targetCenter = (i * spacing) + shear;
            
            // Absolute distance from the center line of the ring
            float ringD = abs(trueD - targetCenter);
            
            // Ring thickness based on roughness (structural complexity)
            float thickness = 2.0 + u_roughness * 4.0;
            
            // Razor sharp cut for the band itself
            float isRing = smoothstep(thickness + 0.5, thickness - 0.5, ringD);
            
            // Deep drop shadow cast from the inner edge
            float shadow = smoothstep(12.0, -2.0, trueD - targetCenter);
            finalColor = mix(finalColor, float3(0.0), shadow * 0.85);
            
            // 1-pixel razor-sharp specular bevel on the very edge
            float bevel = smoothstep(thickness + 0.5, thickness, ringD) * smoothstep(thickness - 1.5, thickness - 0.5, ringD);
            
            // Specular light sweeping across the geometry
            float highlight = sin(theta * (1.0 + i * 0.3) + lightSweepBase) * 0.5 + 0.5;
            float spec = pow(highlight, 24.0); // Mimics machined titanium
            
            // Material definition
            float3 ringColor = mix(u_mutedColor * 0.2, u_dominantColor, 0.3);
            ringColor += u_vibrantColor * spec * (1.5 + u_volatility * 2.5);
            ringColor += mix(u_dominantColor, u_vibrantColor, u_spectralBalance) * bevel * 1.5;
            
            finalColor = mix(finalColor, ringColor, isRing);
        }
        
        finalColor *= smoothstep(0.0, 1.0, trueD);
        finalColor = 1.0 - exp(-finalColor * 2.5); // Premium HDR Tone Mapping
        
        return half4(clamp(finalColor, 0.0, 1.0), 1.0);
    }
"""

@OptIn(UnstableApi::class)
@Composable
fun AetheriumHaloEngine() {
    val context = LocalContext.current
    val coroutineScope = rememberCoroutineScope()
    val audioEngine = remember { AudioTrendEngine() }
    val exoPlayer = remember { setupExoPlayer(context, audioEngine) }
    
    var currentArtBitmap by remember { mutableStateOf<android.graphics.Bitmap?>(null) }
    var mutedColor by remember { mutableStateOf(Color(0xFF0F0F0F)) }
    var dominantColor by remember { mutableStateOf(Color(0xFF2A2A2A)) }
    var vibrantColor by remember { mutableStateOf(Color(0xFFE0E0E0)) }
    
    var kineticTime by remember { mutableFloatStateOf(0f) }
    var smoothedPulse by remember { mutableFloatStateOf(0f) } 
    var pulseVelocity by remember { mutableFloatStateOf(0f) }
    var lastFrameNanos by remember { mutableLongStateOf(0L) }
    
    var isPlaying by remember { mutableStateOf(false) }
    var artSizePx by remember { mutableStateOf(IntSize.Zero) }
    var showDashboard by remember { mutableStateOf(true) } // Dashboard toggle state
    
    val targetSizeDp = 320.dp
    val targetCornerRadiusDp = 36.dp
    val cornerRadiusPx = with(LocalDensity.current) { targetCornerRadiusDp.toPx() }
    val opticalShader = remember { RuntimeShader(AETHERIUM_HALO_SRC) }

    DisposableEffect(exoPlayer) {
        val listener = object : Player.Listener {
            override fun onIsPlayingChanged(p: Boolean) { isPlaying = p }
            override fun onMediaMetadataChanged(m: MediaMetadata) {
                m.artworkData?.let { data ->
                    coroutineScope.launch(Dispatchers.IO) {
                        val bitmap = BitmapFactory.decodeByteArray(data, 0, data.size)
                        Palette.from(bitmap).generate { p ->
                            val fallback = 0xFF444444.toInt()
                            val dom = p?.getDominantColor(fallback) ?: fallback
                            val mut = p?.getDarkMutedColor(dom) ?: dom
                            val vib = p?.getLightVibrantColor(p.getVibrantColor(dom)) ?: dom
                            
                            launch(Dispatchers.Main) {
                                currentArtBitmap = bitmap
                                mutedColor = Color(mut)
                                dominantColor = Color(dom)
                                vibrantColor = Color(vib)
                            }
                        }
                    }
                }
            }
        }
        exoPlayer.addListener(listener)
        onDispose { exoPlayer.release() }
    }

    val filePicker = rememberLauncherForActivityResult(ActivityResultContracts.GetContent()) { uri ->
        uri?.let { exoPlayer.setMediaItem(MediaItem.fromUri(it)); exoPlayer.prepare(); exoPlayer.play() }
    }

    val macroBreath by audioEngine.macroBreath.collectAsState()
    val microPulse by audioEngine.microPulse.collectAsState()
    val spectralBalance by audioEngine.spectralBalance.collectAsState()
    val volatility by audioEngine.volatility.collectAsState()
    val roughness by audioEngine.roughness.collectAsState()

    var radarX by remember { mutableFloatStateOf(0.5f) }
    var radarY by remember { mutableFloatStateOf(0.5f) }

    LaunchedEffect(Unit) {
        while (true) {
            withFrameNanos { frameNanos ->
                if (lastFrameNanos == 0L) lastFrameNanos = frameNanos
                val dt = ((frameNanos - lastFrameNanos) / 1_000_000_000f).coerceIn(0f, 0.033f)
                lastFrameNanos = frameNanos
                
                val stiffness = 35f
                val damping = 2f * sqrt(stiffness) 
                val force = stiffness * (microPulse - smoothedPulse) - damping * pulseVelocity
                pulseVelocity += force * dt
                smoothedPulse += pulseVelocity * dt
                
                val activeAlpha = if (isPlaying) 1f else 0f
                kineticTime += dt * 0.5f * activeAlpha
                
                val targetX = (spectralBalance * 0.7f) + ((1f - (roughness * 5f).coerceIn(0f, 1f)) * 0.3f)
                val targetY = (macroBreath * 0.6f) + (volatility * 0.4f)
                
                radarX += (targetX.coerceIn(0f, 1f) - radarX) * 0.03f 
                radarY += (targetY.coerceIn(0f, 1f) - radarY) * 0.03f
            }
        }
    }

    // Outer Box intercepts double taps to toggle the dashboard
    Box(
        modifier = Modifier
            .fillMaxSize()
            .background(Color.Black)
            .pointerInput(Unit) {
                detectTapGestures(
                    onDoubleTap = { showDashboard = !showDashboard }
                )
            }
    ) {
        
        // LAYER 1: AGSL Background (The Aetherium Halo)
        Canvas(modifier = Modifier.fillMaxSize()) {
            opticalShader.setFloatUniform("u_resolution", size.width, size.height)
            opticalShader.setFloatUniform("u_kineticTime", kineticTime) 
            opticalShader.setFloatUniform("u_macroBreath", macroBreath)
            opticalShader.setFloatUniform("u_microPulse", smoothedPulse.coerceAtLeast(0f)) 
            opticalShader.setFloatUniform("u_spectralBalance", spectralBalance) 
            opticalShader.setFloatUniform("u_volatility", volatility) 
            opticalShader.setFloatUniform("u_roughness", roughness) 
            opticalShader.setFloatUniform("u_boxSizePx", artSizePx.width.toFloat(), artSizePx.height.toFloat())
            opticalShader.setFloatUniform("u_cornerRadiusPx", cornerRadiusPx)
            opticalShader.setFloatUniform("u_mutedColor", mutedColor.red, mutedColor.green, mutedColor.blue)
            opticalShader.setFloatUniform("u_dominantColor", dominantColor.red, dominantColor.green, dominantColor.blue)
            opticalShader.setFloatUniform("u_vibrantColor", vibrantColor.red, vibrantColor.green, vibrantColor.blue)
            
            drawRect(brush = ShaderBrush(opticalShader))
        }

        // LAYER 2: Strict Album Art Anchor (320dp, Centered)
        if (currentArtBitmap != null) {
            Box(
                modifier = Modifier
                    .align(Alignment.Center)
                    .size(targetSizeDp), 
                contentAlignment = Alignment.Center
            ) {
                Image(
                    bitmap = currentArtBitmap!!.asImageBitmap(), 
                    contentDescription = null, 
                    modifier = Modifier
                        .fillMaxSize() 
                        .clip(RoundedCornerShape(targetCornerRadiusDp))
                        .onSizeChanged { artSizePx = it },
                    contentScale = ContentScale.Crop
                )
            }
        }

        // LAYER 3: Animated V6.0 Diagnostic Matrix
        AnimatedVisibility(
            visible = showDashboard,
            enter = fadeIn() + slideInVertically(initialOffsetY = { it }),
            exit = fadeOut() + slideOutVertically(targetOffsetY = { it }),
            modifier = Modifier.align(Alignment.BottomCenter)
        ) {
            Column(
                modifier = Modifier
                    .fillMaxWidth()
                    .background(Color(0xD9080808))
                    .padding(24.dp)
            ) {
                Row(
                    modifier = Modifier.fillMaxWidth().padding(bottom = 16.dp), 
                    horizontalArrangement = Arrangement.SpaceBetween,
                    verticalAlignment = Alignment.CenterVertically
                ) {
                    Column {
                        Text("THE AETHERIUM HALO ENGINE", color = Color.White, style = MaterialTheme.typography.titleMedium, fontWeight = FontWeight.Bold)
                        Text("Crucible v6.0 Diagnostic Matrix (Double-Tap to Hide)", color = Color.Gray, style = MaterialTheme.typography.labelSmall)
                    }
                    Row(horizontalArrangement = Arrangement.spacedBy(8.dp)) {
                        Button(
                            onClick = { if (isPlaying) exoPlayer.pause() else exoPlayer.play() },
                            colors = ButtonDefaults.buttonColors(containerColor = Color.DarkGray)
                        ) {
                            Text(if (isPlaying) "PAUSE" else "PLAY")
                        }
                        Button(
                            onClick = { filePicker.launch("audio/*") }, 
                            colors = ButtonDefaults.buttonColors(containerColor = Color(0xFF2E7D32))
                        ) {
                            Text("OPEN TRACK")
                        }
                    }
                }
                
                Row(modifier = Modifier.fillMaxWidth(), horizontalArrangement = Arrangement.spacedBy(24.dp)) {
                    ExecutiveRadar(xCoord = radarX, yCoord = radarY, dotColor = vibrantColor)
                    
                    Column(modifier = Modifier.weight(1f), verticalArrangement = Arrangement.spacedBy(8.dp)) {
                        
                        Text("MICRO (INTERNAL AGITATION)", color = Color.Gray, fontSize = 10.sp, fontWeight = FontWeight.Bold)
                        TranslationBar("Transients", smoothedPulse.coerceAtLeast(0f), Color(0xFFFF5722), "Band Shear")
                        
                        Spacer(modifier = Modifier.height(2.dp))
                        
                        Text("MESO (FIELD STATE)", color = Color.Gray, fontSize = 10.sp, fontWeight = FontWeight.Bold)
                        TranslationBar("Roughness", roughness, Color(0xFFE040FB), "Band Thickness")
                        TranslationBar("Brightness", spectralBalance, Color(0xFFAEEA00), "Specular Bevel")
                        
                        Spacer(modifier = Modifier.height(2.dp))
                        
                        Text("MACRO (VOLUMETRIC EVOLUTION)", color = Color.Gray, fontSize = 10.sp, fontWeight = FontWeight.Bold)
                        TranslationBar("Bass Energy", macroBreath, Color(0xFF00E5FF), "Ring Spacing")
                        TranslationBar("Kinetic Time", (kineticTime % 1.0f), Color(0xFFFFFFFF), "Light Sweep Phase")
                    }
                }
            }
        }
    }
}

@Composable
fun ExecutiveRadar(xCoord: Float, yCoord: Float, dotColor: Color) {
    Box(
        modifier = Modifier
            .size(160.dp)
            .clip(RoundedCornerShape(12.dp))
            .background(Color(0xFF161616))
    ) {
        Canvas(modifier = Modifier.fillMaxSize()) {
            val w = size.width
            val h = size.height
            drawLine(Color(0xFF333333), Offset(w/2, 0f), Offset(w/2, h), strokeWidth = 2f)
            drawLine(Color(0xFF333333), Offset(0f, h/2), Offset(w, h/2), strokeWidth = 2f)
            
            val dotX = xCoord * w
            val dotY = (1f - yCoord) * h
            drawCircle(color = dotColor.copy(alpha = 0.4f), radius = 12.dp.toPx(), center = Offset(dotX, dotY))
            drawCircle(color = dotColor, radius = 5.dp.toPx(), center = Offset(dotX, dotY))
        }

        Text("SPARSE / RIGID", color = Color(0xFF555555), fontSize = 8.sp, fontWeight = FontWeight.Bold, modifier = Modifier.align(Alignment.TopStart).padding(6.dp))
        Text("DENSE / VIBRANT", color = Color(0xFF555555), fontSize = 8.sp, fontWeight = FontWeight.Bold, modifier = Modifier.align(Alignment.TopEnd).padding(6.dp))
        Text("DEEP / ACTIVE", color = Color(0xFF555555), fontSize = 8.sp, fontWeight = FontWeight.Bold, modifier = Modifier.align(Alignment.BottomStart).padding(6.dp))
        Text("CALM / SHALLOW", color = Color(0xFF555555), fontSize = 8.sp, fontWeight = FontWeight.Bold, modifier = Modifier.align(Alignment.BottomEnd).padding(6.dp))
    }
}

@Composable
fun TranslationBar(audioLabel: String, value: Float, color: Color, visualResultLabel: String) {
    Row(verticalAlignment = Alignment.CenterVertically, modifier = Modifier.fillMaxWidth()) {
        Text(audioLabel, color = Color.White, style = MaterialTheme.typography.labelSmall, modifier = Modifier.width(90.dp))
        Canvas(modifier = Modifier.height(4.dp).weight(1f).clip(RoundedCornerShape(2.dp))) {
            drawRect(color = Color(0xFF222222))
            drawRect(color = color, size = size.copy(width = size.width * value.coerceIn(0f, 1f)))
        }
        Text("→ $visualResultLabel", color = Color.Gray, style = MaterialTheme.typography.labelSmall, textAlign = TextAlign.End, modifier = Modifier.width(130.dp).padding(start = 8.dp))
    }
}

@UnstableApi
fun setupExoPlayer(context: Context, audioEngine: AudioTrendEngine): ExoPlayer {
    val teeProcessor = TeeAudioProcessor(audioEngine)
    val audioSink = DefaultAudioSink.Builder().setAudioProcessors(arrayOf(teeProcessor)).build()
    val renderersFactory = object : DefaultRenderersFactory(context) {
        override fun buildAudioSink(c: Context, f: Boolean, a: Boolean, e: Boolean) = audioSink
    }
    return ExoPlayer.Builder(context, renderersFactory).build().apply { repeatMode = Player.REPEAT_MODE_ONE }
}
