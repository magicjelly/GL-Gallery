// ==========================================
// 1. HARDWARE-ALIGNED ACOUSTIC ENVELOPE
// ==========================================
@UnstableApi
class AudioTrendEngine : TeeAudioProcessor.AudioBufferSink {
    var audioGain: Float = 25.0f 
    
    private val _realTimePeak = MutableStateFlow(0f)
    val realTimePeak: StateFlow<Float> = _realTimePeak 

    private val _macroBreath = MutableStateFlow(0f)
    val macroBreath: StateFlow<Float> = _macroBreath 

    private val _microPulse = MutableStateFlow(0f)
    val microPulse: StateFlow<Float> = _microPulse

    private var currentAfterglow = 0f
    private var currentPulse = 0f
    private var prevFiltered = 0f

    override fun flush(sampleRateHz: Int, channelCount: Int, encoding: Int) {
        currentAfterglow = 0f; currentPulse = 0f; prevFiltered = 0f
        _realTimePeak.value = 0f; _macroBreath.value = 0f; _microPulse.value = 0f
    }

    override fun handleBuffer(buffer: ByteBuffer) {
        val limit = buffer.limit()
        if (limit == 0) return
        var sumSq = 0.0
        buffer.position(0)
        while (buffer.position() < limit - 1) {
            val raw = buffer.short.toFloat() / Short.MAX_VALUE
            val filtered = prevFiltered + 0.05f * (raw - prevFiltered)
            prevFiltered = filtered
            sumSq += (filtered * filtered)
        }
        val rawRms = sqrt(sumSq / (limit / 2)).toFloat()
        val compressedEnergy = (1.0f - exp(-rawRms * audioGain)).coerceIn(0f, 1f)
        
        _realTimePeak.value = compressedEnergy
        
        // Massive inertia for heavy fluid rotation (0.985 damping)
        currentAfterglow = (currentAfterglow * 0.985f) + (compressedEnergy * 0.015f)
        _macroBreath.value = currentAfterglow.coerceIn(0f, 1f)
        
        // Reactive acoustic pressure for volumetric extrusion (0.80 damping)
        currentPulse = (currentPulse * 0.8f) + (compressedEnergy * 0.2f)
        _microPulse.value = currentPulse.coerceIn(0f, 1f)
    }
}

// ==========================================
// 2. KINETIC TOPOLOGICAL VOLUME SHADER
// ==========================================
const val TOPOLOGICAL_VOLUME_SHADER_SRC = """
    uniform float2 u_resolution;
    uniform float u_macroBreath; 
    uniform float u_microPulse;  
    uniform float u_time;    
    
    uniform float3 u_mutedColor;
    uniform float3 u_dominantColor;
    uniform float3 u_vibrantColor;

    // Sub-pixel stochastic dither seed
    float2 hash22(float2 p) {
        float3 p3 = fract(float3(p.xyx) * float3(.1031, .1030, .0973));
        p3 += dot(p3, p3.yzx + 33.33);
        return fract((p3.xx+p3.yz)*p3.zy);
    }

    // Smooth volumetric noise
    float noise(float2 x) {
        float2 i = floor(x);
        float2 f = fract(x);
        f = f * f * (3.0 - 2.0 * f);
        float res = mix(
            mix(dot(hash22(i + float2(0.0, 0.0)), f - float2(0.0, 0.0)),
                dot(hash22(i + float2(1.0, 0.0)), f - float2(1.0, 0.0)), f.x),
            mix(dot(hash22(i + float2(0.0, 1.0)), f - float2(0.0, 1.0)),
                dot(hash22(i + float2(1.0, 1.0)), f - float2(1.0, 1.0)), f.x), f.y);
        return res * 0.5 + 0.5;
    }

    // Perceptual color mixing proxy (prevents muddy desaturation in transitions)
    half3 perceptualMix(half3 c1, half3 c2, float t) {
        // Squaring the inputs roughly approximates a linear working space
        half3 l1 = c1 * c1;
        half3 l2 = c2 * c2;
        return sqrt(mix(l1, l2, t));
    }

    float getTopologicalDensity(float2 uv, float time, float breath) {
        // Multi-octave domain warping creates the uncountable fluid forms
        float2 q = float2(
            noise(uv * 1.5 + time * 0.12),
            noise(uv * 1.8 - time * 0.15)
        );
        
        float2 r = float2(
            noise(uv * 3.0 + q * 1.5 + time * 0.2),
            noise(uv * 3.2 + q * 2.0 - time * 0.18)
        );
        
        // Fluid rotation driven by macro acoustic breath
        float density = noise(uv * 2.5 + r * (1.5 + breath * 0.8) - time * 0.1);
        return density;
    }

    half4 main(float2 fragCoord) {
        float2 uv = fragCoord / u_resolution;
        float aspect = u_resolution.x / u_resolution.y;
        uv.x *= aspect;
        
        float2 screenCenter = float2(0.5 * aspect, 0.5);
        float distFromCenter = distance(uv, screenCenter);
        
        // Central exclusion zone guarantees absolute UI legibility
        float centerProtect = smoothstep(0.15, 0.65, distFromCenter); 
        
        // --- 1. THE VOLUMETRIC DOMAIN ---
        float baseDensity = getTopologicalDensity(uv, u_time, u_macroBreath);
        
        // --- 2. KINETIC EXTRUSION (Depth Contrast) ---
        // The acoustic micro pulse physically extrudes the dense areas of the fluid forward.
        // It acts as a multiplier, pulling peaks out of the background void.
        float extrusionForce = 1.0 + (u_microPulse * 1.8 * centerProtect);
        float activeDensity = baseDensity * extrusionForce;
        
        // --- 3. PERCEPTUAL COLOR MAPPING ---
        half3 color = half3(u_mutedColor);
        
        // Deep background transitions to dominant mid-volume
        float t1 = smoothstep(0.3, 0.7, activeDensity);
        color = perceptualMix(color, half3(u_dominantColor), t1);
        
        // Mid-volume extrudes into vibrant peaks based on acoustic pressure
        float t2 = smoothstep(0.6, 1.2, activeDensity);
        color = perceptualMix(color, half3(u_vibrantColor), t2);
        
        // Deepen the shadows to establish massive 3D depth without requiring reflections
        color *= smoothstep(0.0, 0.5, activeDensity);
        
        // Edge vignette
        color *= smoothstep(1.3, 0.4, distFromCenter); 
        
        // --- 4. HARDWARE DITHERING ---
        // Extremely subtle high-frequency noise physically breaks OLED color banding.
        float dither = (hash22(fragCoord).x - 0.5) * 0.015;
        color += half3(dither);
        
        return half4(clamp(color, 0.0, 1.0), 1.0);
    }
"""

@OptIn(UnstableApi::class)
@Composable
fun TopologicalVolumeMediaPrototype() {
    val context = LocalContext.current
    val audioEngine = remember { AudioTrendEngine() }
    val exoPlayer = remember { setupExoPlayer(context, audioEngine) }
    
    var currentArtBitmap by remember { mutableStateOf<android.graphics.Bitmap?>(null) }
    var mutedColor by remember { mutableStateOf(Color(0xFF1A1A1A)) }
    var dominantColor by remember { mutableStateOf(Color(0xFF424242)) }
    var vibrantColor by remember { mutableStateOf(Color(0xFF9E9E9E)) }
    
    var engineTime by remember { mutableFloatStateOf(0f) }
    var isPresentationMode by remember { mutableStateOf(false) }
    var isPlaying by remember { mutableStateOf(false) }

    DisposableEffect(exoPlayer) {
        val listener = object : Player.Listener {
            override fun onIsPlayingChanged(p: Boolean) { isPlaying = p }
            override fun onMediaMetadataChanged(m: MediaMetadata) {
                m.artworkData?.let { data ->
                    val bitmap = BitmapFactory.decodeByteArray(data, 0, data.size)
                    currentArtBitmap = bitmap
                    Palette.from(bitmap).generate { p ->
                        // Fallbacks ensure the volume maintains deep contrast
                        val fallbackDark = 0xFF121212.toInt()
                        val fallbackMid = 0xFF37474F.toInt()
                        val fallbackLight = 0xFF78909C.toInt()
                        
                        mutedColor = Color(p?.getDarkMutedColor(fallbackDark) ?: fallbackDark)
                        dominantColor = Color(p?.getDominantColor(fallbackMid) ?: fallbackMid)
                        vibrantColor = Color(p?.getLightVibrantColor(p.getVibrantColor(fallbackLight)) ?: fallbackLight)
                    }
                }
            }
        }
        exoPlayer.addListener(listener)
        onDispose { exoPlayer.release() }
    }

    val filePicker = rememberLauncherForActivityResult(ActivityResultContracts.GetContent()) { uri ->
        uri?.let { exoPlayer.setMediaItem(MediaItem.fromUri(it)); exoPlayer.prepare(); exoPlayer.play() }
    }

    val macroBreath by audioEngine.macroBreath.collectAsState()
    val microPulse by audioEngine.microPulse.collectAsState()
    val opticalShader = remember { RuntimeShader(TOPOLOGICAL_VOLUME_SHADER_SRC) }

    var activeStateAlpha by remember { mutableFloatStateOf(1f) }

    LaunchedEffect(Unit) {
        while (true) {
            withFrameMillis { 
                val targetMult = if (isPlaying) 1f else 0.2f
                activeStateAlpha = (0.02f * targetMult) + (0.98f * activeStateAlpha)
                engineTime += 0.003f * activeStateAlpha
            }
        }
    }

    Column(modifier = Modifier.fillMaxSize().background(Color.Black)) {
        Box(modifier = Modifier.weight(if (isPresentationMode) 1f else 0.55f).fillMaxWidth(), contentAlignment = Alignment.Center) {
            Canvas(modifier = Modifier.fillMaxSize()) {
                opticalShader.setFloatUniform("u_resolution", size.width, size.height)
                opticalShader.setFloatUniform("u_time", engineTime)
                opticalShader.setFloatUniform("u_macroBreath", macroBreath * activeStateAlpha)
                opticalShader.setFloatUniform("u_microPulse", microPulse * activeStateAlpha)
                
                opticalShader.setFloatUniform("u_mutedColor", mutedColor.red, mutedColor.green, mutedColor.blue)
                opticalShader.setFloatUniform("u_dominantColor", dominantColor.red, dominantColor.green, dominantColor.blue)
                opticalShader.setFloatUniform("u_vibrantColor", vibrantColor.red, vibrantColor.green, vibrantColor.blue)
                
                drawRect(brush = ShaderBrush(opticalShader))
            }
            if (currentArtBitmap != null) {
                Image(bitmap = currentArtBitmap!!.asImageBitmap(), contentDescription = null, 
                    modifier = Modifier.size(280.dp).clip(RoundedCornerShape(36.dp)))
            }
        }

        AnimatedVisibility(visible = !isPresentationMode) {
            Column(modifier = Modifier.fillMaxWidth().background(Color(0xFF080808)).padding(24.dp), horizontalAlignment = Alignment.CenterHorizontally) {
                Row(modifier = Modifier.fillMaxWidth(), horizontalArrangement = Arrangement.SpaceBetween, verticalAlignment = Alignment.CenterVertically) {
                    Text("TOPOLOGICAL VOLUME", color = Color.White, style = MaterialTheme.typography.titleSmall)
                    Button(onClick = { filePicker.launch("audio/*") }, colors = ButtonDefaults.buttonColors(containerColor = Color(0xFF2E7D32))) {
                        Text("OPEN TRACK", style = MaterialTheme.typography.labelSmall)
                    }
                }
                Spacer(modifier = Modifier.height(20.dp))
                
                TelemetryBar("VOLUMETRIC SWELL (PULSE)", microPulse, Color(0xFFFF5722))
                TelemetryBar("FLUID INERTIA (BREATH)", macroBreath, Color(0xFF00E5FF))
                
                Spacer(modifier = Modifier.height(24.dp))
                Row(horizontalArrangement = Arrangement.spacedBy(24.dp)) {
                    FloatingActionButton(onClick = { exoPlayer.seekTo(0); exoPlayer.play() }, containerColor = Color.DarkGray, shape = CircleShape, modifier = Modifier.size(48.dp)) {
                        Icon(Icons.Default.Refresh, null, tint = Color.White)
                    }
                    FloatingActionButton(onClick = { if (isPlaying) exoPlayer.pause() else exoPlayer.play() }, containerColor = Color.White, shape = CircleShape) {
                        Icon(if (isPlaying) Icons.Default.Close else Icons.Default.PlayArrow, null, tint = Color.Black)
                    }
                }
            }
        }
        
        Box(modifier = Modifier.fillMaxWidth().padding(8.dp), contentAlignment = Alignment.Center) {
            TextButton(onClick = { isPresentationMode = !isPresentationMode }) {
                Text(if (isPresentationMode) "VIEW TUNER" else "PRESENTATION MODE", color = Color.Gray)
            }
        }
    }
}

@Composable
fun TelemetryBar(label: String, value: Float, color: Color) {
    Row(verticalAlignment = Alignment.CenterVertically, modifier = Modifier.padding(vertical = 4.dp)) {
        Text(label, color = Color.Gray, style = MaterialTheme.typography.labelSmall, modifier = Modifier.width(220.dp))
        Canvas(modifier = Modifier.height(8.dp).weight(1f).clip(RoundedCornerShape(4.dp))) {
            drawRect(color = Color.DarkGray); drawRect(color = color, size = size.copy(width = size.width * value))
        }
    }
}

@UnstableApi
fun setupExoPlayer(context: Context, audioEngine: AudioTrendEngine): ExoPlayer {
    val teeProcessor = TeeAudioProcessor(audioEngine)
    val audioSink = DefaultAudioSink.Builder().setAudioProcessors(arrayOf(teeProcessor)).build()
    val renderersFactory = object : DefaultRenderersFactory(context) {
        override fun buildAudioSink(c: Context, f: Boolean, a: Boolean, e: Boolean) = audioSink
    }
    return ExoPlayer.Builder(context, renderersFactory).build().apply { repeatMode = Player.REPEAT_MODE_ONE }
}
